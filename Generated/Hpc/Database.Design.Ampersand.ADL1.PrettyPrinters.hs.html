   12 prettyPrint :: Pretty a => a -> String
   13 prettyPrint x = displayS (renderPretty rfrac col_width doc) ""
   14         where col_width = 120
   15               rfrac = 0.4
   16               doc = pretty x
   17 
   18 (<~>) :: Pretty b => Doc -> b -> Doc
   19 (<~>) a b = a <+> pretty b
   20 
   21 (<+\>) :: Doc -> Doc -> Doc
   22 (<+\>) a b = a <$$> b
   23 
   24 (<~\>) :: Pretty b => Doc -> b -> Doc
   25 (<~\>) a b = a <+\> pretty b
   26 
   27 perline :: Pretty a => [a] -> Doc
   28 perline = vsep . map pretty
   29 
   30 perlinePrefix :: Pretty a => String -> [a] -> Doc
   31 perlinePrefix pref xs = vsep $ map addPrefix xs
   32            where addPrefix x = text pref <+> pretty x
   33

   37 quote :: String -> Doc
   38 quote = dquotes.text.escapeAll
   39         where escapeAll = escapeQuote.escapeBreaklines.escapeSlash
   40               escapeQuote = escape "\""
   41               escapeBreaklines = replace "\n" "\\n"
   42               escapeSlash = escape "\\"
   43               escape x = replace x ("\\" ++ x)
   44 
   45 quoteAll :: [String] -> [Doc]
   46 quoteAll = map quote
   47 
   48 quotePurpose :: String -> Doc
   49 quotePurpose p = text "{+" </> escapeExpl p </> text "-}"
   50         where escapeExpl = text.escapeCommentStart.escapeLineComment.escapeExplEnd
   51               escapeCommentStart = escape "{-"
   52               escapeLineComment = escape "--"
   53               escapeExplEnd = escape "-}"
   54               escape x = replace x (intersperse ' ' x)
   55 
   56 isId :: String -> Bool
   57 isId a = not (null a) && all isIdChar a && isFirstIdChar(head a) && a `notElem` keywords
   58        where isFirstIdChar x = elem x $ "_"++['a'..'z']++['A'..'Z']
   59              isIdChar x = isFirstIdChar x || elem x ['0'..'9']
   60 
   61 isUpperId :: String -> Bool
   62 isUpperId xs = isId xs && head xs `elem` ['A'..'Z']
   63 
   64 maybeQuote :: String -> Doc
   65 maybeQuote a = if isId a then text a else quote a
   66 
   67 -- adds quotes unless it's an upper identifier
   68 quoteConcept :: String -> Doc
   69 quoteConcept a = if isUpperId a then text a else quote a
   70 
   71 prettyhsep :: Pretty a => [a] -> Doc
   72 prettyhsep = hsep . map pretty
   73 
   74 commas :: [Doc] -> Doc
   75 commas = encloseSep empty empty comma
   76 
   77 listOf :: Pretty a => [a] -> Doc
   78 listOf = commas . map pretty
   79 
   80 prettyPair :: Paire -> Doc
   81 prettyPair (Paire src trg) = quote src <+> text "*" <+> quote trg
   82 
   83 listOfLists :: [[String]] -> Doc
   84 listOfLists xs = commas $ map (hsep.quoteAll) xs
   85 
   86 separate :: Pretty a => String -> [a] -> Doc
   87 separate d xs = encloseSep empty empty (text d) $ map pretty xs

   94 labelArgs :: [[String]] -> Doc
   95 labelArgs args = if null args || all null args
   96                  then empty
   97                  else braces $ listOfLists args

   99 prettyLabel :: String -> [[String]] -> Doc
  100 prettyLabel nm strs = maybeQuote nm <+> labelArgs strs

  102 instance Pretty P_Context where
  103     pretty (PCtx nm _ lang markup thms pats rs ds cs ks rrules rrels vs gs ifcs ps pops sql php metas) =
  104                text "CONTEXT"
  105                <+> quoteConcept nm
  106                <~> lang
  107                <~> markup
  108                <+\> perline metas
  109                <+\> themes
  110                <+\> perline ps
  111                <+\> perline pats
  112                <+\> perline rs
  113                <+\> perline ds
  114                <+\> perline cs
  115                <+\> perline ks
  116                <+\> perline rrules
  117                <+\> perline rrels
  118                <+\> perline vs
  119                <+\> perline gs
  120                <+\> perline ifcs
  121                <+\> perline pops
  122                <+\> perlinePrefix "SQLPLUG" sql
  123                <+\> perlinePrefix "PHPPLUG" php
  124                <+\> text "ENDCONTEXT"
  125              where themes | null thms = empty
  126                           | otherwise = text "THEMES" <+> commas (map quoteConcept thms)

  146 instance Pretty P_Pattern where
  147     pretty (P_Pat _ nm rls gns dcs rruls rrels cds ids vds xps pop _) =
  148           text keyword
  149           <+>  quoteConcept nm
  150           <+\> perline rls

  163 instance Pretty P_Declaration where
  164     pretty (P_Sgn nm sign prps pragma mean popu _ plug) =
  165         text "RELATION" <+> text nm <~> sign <+> props <+> byplug <+\> pragmas <+\> prettyhsep mean <+\> content
  166         where props   = if prps == [Sym, Asy] then text "[PROP]"
  167                         else text "[" <> listOf prps <> text "]"
  168               byplug  | plug        = text "BYPLUG"
  169                       | otherwise   = empty
  170               pragmas | null pragma = empty
  171                       | otherwise   = text "PRAGMA" <+> hsep (map quote pragma)
  172               content | null popu   = empty
  173                       | otherwise   = text "=\n[" <+> commas (map prettyPair popu) <+> text "]"

  175 instance Pretty a => Pretty (Term a) where
  176    pretty p = case p of
  177        Prim a -> pretty a
  178        -- level 0 (rule)
  179        PEqu _ t1 t2 -> two t1 t2 "="
  180        PImp _ t1 t2 -> two t1 t2 " |- "
  181        -- level 1
  182        PIsc _ t1 t2 -> two t1 t2 "/\\"
  183        PUni _ t1 t2 -> two t1 t2 "\\/"
  184        -- level 2
  185        PDif _ t1 t2 -> two t1 t2 "-"
  186        -- level 3
  187        PLrs _ t1 t2 -> two t1 t2 "/"
  188        PRrs _ t1 t2 -> two t1 t2 "\\"
  189        PDia _ t1 t2 -> two t1 t2 "<>"
  190        -- level 4
  191        PCps _ t1 t2 -> two t1 t2 ";"
  192        PRad _ t1 t2 -> two t1 t2 "!"
  193        PPrd _ t1 t2 -> two t1 t2 "#"
  194        -- level 5
  195        PKl0 _ t -> pos t "*"
  196        PKl1 _ t -> pos t "+"
  197        PFlp _ t -> pos t "~"
  198        PCpl _ t -> pre t " -" -- a double dash can happen when combined with PDif, therefore the extra space
  199        -- level 6
  200        PBrk _ t -> parens $ pretty t
  201        
  202        where pos t op     = pretty t <> text op
  203              pre t op     = text op <> pretty t
  204              two t1 t2 op = pretty t1 <> text op <> pretty t2

  206 instance Pretty TermPrim where
  207     pretty p = case p of
  208         PI _ -> text "I"
  209         Pid _ concept -> text "I[" <> pretty concept <> text "]"
  210         Patm _ str (Just concept) -> singleQuote str <> text "[" <> pretty concept <> text "]"
  211         Patm _ str Nothing -> singleQuote str
  212         PVee _ -> text "V"
  213         Pfull _ s1 s2 -> text "V" <~> P_Sign s1 s2
  214         PNamedR rel -> pretty rel
  215       where singleQuote = squotes . text

  241 instance Pretty ConceptDef where
  242     pretty (Cd _ cpt plug def typ ref _) -- from, the last argument, is not used in the parser
  243         = text "CONCEPT" <+> quoteConcept cpt <+> (if plug then text "BYPLUG" else empty)
  244                <+> quote def <+> type_ <+> maybeText ref
  245         where type_ = if null typ then empty
  246                       else text "TYPE" <+> quote typ
  247               maybeText txt = if null txt then empty
  248                               else quote txt

  250 instance Pretty P_Population where
  251     pretty p = case p of
  252                 P_RelPopu _ nm    cs -> text "POPULATION" <+> maybeQuote (takeQuote nm)        <+> text "CONTAINS" <+> contents cs
  253                 P_TRelPop _ nm tp cs -> text "POPULATION" <+> maybeQuote (takeQuote nm) <~> tp <+> text "CONTAINS" <+> contents cs

  257 instance Pretty P_Interface where
  258     pretty (P_Ifc name klass prms args roles obj _ _) =
  259         text "INTERFACE" <+> maybeQuote name <+> class_
  260                <+> params <+> labelArgs args <+> iroles
  261                <+> text ":" <~\> obj_ctx obj <~> obj_msub obj
  262                  where class_ = case klass of
  263                                      Nothing  -> empty
  264                                      Just str -> text "CLASS" <+> quoteConcept str
  265                        params = if null prms then empty
  266                                 else parens $ listOf prms
  267                        iroles = if null roles then empty
  268                                 else text "FOR" <+> listOf roles

  270 instance Pretty a => Pretty (P_ObjDef a) where
  271     pretty (P_Obj nm _ ctx mView msub strs) =
  272         prettyLabel nm strs <+> text ":"
  273                  <~> ctx <+> view mView <~> msub
  274         where view Nothing  = empty
  275               view (Just v) = text ("<" ++ v ++ ">")

  296 instance Pretty a => Pretty (P_ViewD a) where
  297     pretty (P_Vd _ lbl cpt True Nothing ats) = -- legacy syntax

  299                     <~> cpt <+> parens (listOf ats)
