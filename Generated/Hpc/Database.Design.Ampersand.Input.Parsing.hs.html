   27 -- Parse an Ampersand file and all transitive includes
   28 parseADL ::  Options -> FilePath -> IO (Guarded P_Context)
   29 parseADL opts filePath =
   30   whenCheckedIO (parseSingleADL opts filePath) $ \(ctxt, filePaths) ->
   31     whenCheckedIO (parseADLs opts [filePath] filePaths) $ \ctxts ->
   32       return $ Checked $ foldl mergeContexts ctxt ctxts
   33 
   34 parseADLs :: Options -> [FilePath] -> [FilePath] -> IO (Guarded [P_Context])
   35 parseADLs _    _               []        = return $ Checked []
   36 parseADLs opts parsedFilePaths filePaths =
   37  do { let filePathsToParse = nub filePaths \\ parsedFilePaths
   38     ; whenCheckedIO (sequenceA <$> mapM (parseSingleADL opts) filePathsToParse) $ \ctxtNewFilePathss ->
   39        do { let (ctxts, newFilessToParse) = unzip ctxtNewFilePathss
   40           ; whenCheckedIO (parseADLs opts (parsedFilePaths ++ filePaths) $ concat newFilessToParse) $ \ctxts' ->
   41               return $ Checked $ ctxts ++ ctxts'
   42           }
   43     }

   45 -- Parse an Ampersand file, but not its includes (which are simply returned as a list)
   46 parseSingleADL :: Options -> FilePath -> IO (Guarded (P_Context, [FilePath]))
   47 parseSingleADL opts filePath =
   48  do { verboseLn opts $ "Reading file " ++ filePath
   49     ; mFileContents <- readUTF8File filePath
   50     ; case mFileContents of
   51         Left err -> return $ makeError ("ERROR reading file " ++ filePath ++ ":\n" ++ err)
   52         Right fileContents ->
   53              whenCheckedIO (return $ parseCtx filePath fileContents) $ \(ctxts, relativePaths) -> 
   54                    do filePaths <- mapM normalizePath relativePaths
   55                       return (Checked (ctxts, filePaths))
   56     }
   57  where normalizePath relativePath = canonicalizePath $ takeDirectory filePath </> relativePath

   59 parseErrors :: Lang -> ParseError -> [CtxError]
   60 parseErrors lang err = [PE (Message msg)]
   61                 where msg :: String
   62                       msg = show (errorPos err) ++ ":" ++ showLang lang (errorMessages err)
   63                       showLang :: Lang -> [Message] -> String
   64                       showLang English = showErrorMessages "or" "unknown parse error"   "expecting" "unexpected" "end of input"
   65                       showLang Dutch   = showErrorMessages "of" "onbekende parsingfout" "verwacht"  "onverwacht" "einde van de invoer"

   67 parse :: AmpParser a -> FilePath -> [Token] -> Guarded a
   68 parse p fn ts =
   69       -- runP :: Parsec s u a -> u -> FilePath -> s -> Either ParseError a 
   70     case runP p pos fn ts of
   71         --TODO: Add language support to the parser errors
   72         Left err -> Errors $ parseErrors English err
   73         Right a -> Checked a
   74     where pos | null ts   = initPos fn
   75               | otherwise = tokPos (head ts)

   81 runParser :: AmpParser a -> Filename -> String -> Guarded a
   82 runParser parser filename input =
   83   -- lexer :: [Options] -> String -> [Char] -> Either LexerError ([Token], [LexerWarning])
   84   --TODO: Give options to the lexer
   85   let lexed = lexer [] filename input
   86   in case lexed of
   87     Left err -> Errors $ lexerErrors err
   88     --TODO: Do something with the warnings. The warnings cannot be shown with the current Guarded data type
   89     Right (tokens, _)  -> whenChecked (parse parser filename tokens) Checked

   91 -- In order to read derivation rules, we use the Ampersand parser.
   92 -- Since it is applied on static code only, error messagea may be produced as fatals.
   93 parseRule :: String -> Term TermPrim
   94 parseRule str
   95    = case  runParser pRule "inside Haskell code" str of
   96        Checked result -> result
   97        Errors  msg    -> fatal 274 ("Parse errors in "++str++":\n   "++show msg)

   99 -- | Parse isolated ADL1 expression strings
  100 parseADL1pExpr :: String            -- ^ The string to be parsed
  101                -> FilePath          -- ^ The name of the file (used for error messages)
  102                -> Either String (Term TermPrim)  -- ^ The result: Either an error message,  or a good result
  103 parseADL1pExpr str fn =
  104   case runParser pTerm fn str of
  105       Checked result -> Right result
  106       Errors  msg    -> Left $ "Parse errors:\n"++show msg
