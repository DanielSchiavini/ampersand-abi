   12 -- | use a suitable format to read generated strings. if you have just normal text, ReST is fine.
   13 -- | defaultPandocReader getOpts should be used on user-defined strings.
   14 string2Blocks :: PandocFormat -> String -> [Block]
   15 string2Blocks defaultformat str
   16  = case blocks of             -- WHY (SJ, dec 7th, 2011): What is the point of changing Para into Plain?
   17     [Para is] -> [Plain is]   -- BECAUSE (SJ, dec 7th, 2011): The table of relations in the LaTeX output of ChapterDataAnalysis gives errors when LaTeX is run, because Para generates a newline that LaTeX cannot cope with.
   18     _         -> blocks       --                              However, this Para is generated by Pandoc, so I'm wondering whether the mistake is in Pandoc? Anyway, this solution is a dirty fix, which I don't like.
   19    where
   20      Pandoc _ blocks = thePandocParser (removeCRs str')  --TODO: catch errors. (See ticket #460 Wrong error message in FSPEC generator when parsing )
   21      removeCRs :: String -> String
   22      removeCRs [] = []
   23      removeCRs ('\r' :'\n' : xs) = '\n' : removeCRs xs
   24      removeCRs (c:xs) = c:removeCRs xs
   25      (thePandocParser,str') = whatParser2UseOnWhatString
   26      whatParser2UseOnWhatString :: (String -> Pandoc,String)
   27      whatParser2UseOnWhatString -- = (readRST, str)
   28         | markDownPrefix `isPrefixOf` str = (readMarkdown def, drop (length markDownPrefix) str)
   29         | reSTPrefix     `isPrefixOf` str = (readRST      def, drop (length reSTPrefix)     str)
   30         | hTMLPrefix     `isPrefixOf` str = (readHtml     def, drop (length hTMLPrefix)     str)
   31          --stateParseRaw=True e.g. such that "\ref{something}" is not read as "\{something\}". with parse raw it's read as inline latex
   32          --maybe html should be parsed raw too...
   33         | laTeXPrefix    `isPrefixOf` str = (readLaTeX    def, drop (length laTeXPrefix)    str)
   34         | otherwise   = case defaultformat of
   35                                Markdown  -> (readMarkdown def, str)
   36                                ReST      -> (readRST      def, str)
   37                                HTML      -> (readHtml     def, str)
   38                                LaTeX     -> (readLaTeX    def, str)
   39        where markDownPrefix = makePrefix Markdown
   40              reSTPrefix     = makePrefix ReST
   41              hTMLPrefix     = makePrefix HTML
   42              laTeXPrefix    = makePrefix LaTeX
   43 
   44 makePrefix :: PandocFormat -> String
   45 makePrefix format = ":"++show format++":"

   47 -- | write [Block] as String in a certain format using defaultWriterOptions
   48 blocks2String :: PandocFormat -> Bool -> [Block] -> String
   49 blocks2String format writeprefix ec
   50  = [c | c<-makePrefix format,writeprefix]
   51    --you cannot unwords lines for all writers, because white lines have a meaning in LaTeX i.e. \par
   52    --if your application of blocks2String may not have line breaks, then unwords lines there
   53    ++ {- unwords -} ( {-lines $ -} writer def (Pandoc nullMeta ec))
   54    where writer = case format of
   55             Markdown  -> writeMarkdown
   56             ReST      -> writeRST
   57             HTML      -> writeHtmlString
   58             LaTeX     -> writeLaTeX
