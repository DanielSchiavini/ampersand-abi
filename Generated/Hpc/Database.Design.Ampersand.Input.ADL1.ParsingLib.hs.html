   45 (<??>) :: AmpParser a -> AmpParser (a -> a) -> AmpParser a
   46 p <??> q = (\x f -> f x) CA.<$> p CA.<*> (q `opt` id)

   49 pIsThere p = (True <$ p) `opt` False

   79 -----------------------------------------------------------
   80 -- Token parsers
   81 -----------------------------------------------------------
   82 
   83 check :: (Lexeme -> Maybe a) -> AmpParser a
   84 check predicate = tokenPrim showTok nextPos matchTok
   85   where  -- Token pretty-printing function
   86          showTok :: Token -> String
   87          showTok (Tok lx _)   = show lx
   88          -- Next position calculating function
   89          nextPos :: SourcePos -> Token -> [Token] -> SourcePos
   90          nextPos pos _ [] = pos
   91          nextPos _ _ (Tok _ (FilePos file line col):_) = newPos file line col
   92          -- ^ Matching function for the token to parse.
   93          matchTok (Tok l _) = predicate l
   94 
   95 match :: Lexeme -> AmpParser String
   96 match lx = check (\lx' -> if lx == lx' then Just (lexemeText lx) else Nothing) <?> show lx
   97 
   98 --- Conid ::= UpperChar (Char | '_')*
   99 pConid :: AmpParser String
  100 pConid = check (\lx -> case lx of { LexConId s -> Just s; _ -> Nothing })
  101 
  102 --- String ::= '"' Any* '"'
  103 --- StringListSemi ::= String (';' String)*
  104 pString :: AmpParser String
  105 pString = check (\lx -> case lx of { LexString s -> Just s; _ -> Nothing })
  106 
  107 --- Expl ::= '{+' Any* '-}'
  108 pExpl :: AmpParser String
  109 pExpl = check (\lx -> case lx of { LexExpl s -> Just s; _ -> Nothing })
  110 
  111 --- Varid ::= (LowerChar | '_') (Char | '_')*
  112 pVarid :: AmpParser String
  113 pVarid = check (\lx -> case lx of { LexVarId s -> Just s; _ -> Nothing })
  114 
  115 --- Atom ::= "'" Any* "'"
  116 pAtom :: AmpParser String

  119 -----------------------------------------------------------
  120 -- Integers
  121 -----------------------------------------------------------
  122 
  123 pNumber :: Int -> AmpParser String
  124 pNumber nr = match (LexDecimal nr) <|> match (LexHex nr) <|> match (LexOctal nr)
  125 
  126 pInteger :: AmpParser Int
  127 pInteger = check isNr
  128     where isNr (LexDecimal i) = Just i
  129           isNr (LexHex i)     = Just i
  130           isNr (LexOctal i)   = Just i
  131           isNr _              = Nothing
  132 
  133 pZero :: AmpParser String
  134 pZero = pNumber 0
  135 
  136 pOne :: AmpParser String
  137 pOne = pNumber 1

  143 -- matches special characters
  144 pSpec :: Char -> AmpParser String
  145 pSpec sym = match (LexSymbol sym)
  146 
  147 pComma :: AmpParser String
  148 pComma  = pSpec ','
  149 
  150 pParens :: AmpParser a -> AmpParser a
  151 pParens parser = pSpec '(' CA.*> parser CA.<* pSpec ')'
  152 
  153 pBraces :: AmpParser a -> AmpParser a
  154 pBraces parser = pSpec '{' CA.*> parser CA.<* pSpec '}'
  155 
  156 pBrackets :: AmpParser a -> AmpParser a
  157 pBrackets parser = pSpec '[' CA.*> parser CA.<* pSpec ']'
  158 
  159 pChevrons :: AmpParser a -> AmpParser a
  160 pChevrons parser = pSpec '<' CA.*> parser CA.<* pSpec '>'

  166 posOrigin :: Show a => a -> SourcePos -> Origin
  167 posOrigin sym p = FileLoc (FilePos (sourceName p) (sourceLine p) (sourceColumn p)) (show sym)
  168 
  169 currPos :: AmpParser Origin
  170 currPos = posOf $ return ()
  171 
  172 posOf :: Show a => AmpParser a -> AmpParser Origin
  173 posOf parser = do { pos <- getPosition; a <- parser; return (posOrigin a pos) }
  174 
  175 valPosOf :: Show a => AmpParser a -> AmpParser (a, Origin)
  176 valPosOf parser = do { pos <- getPosition; a <- parser; return (a, posOrigin a pos) }
