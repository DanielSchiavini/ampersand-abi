   13 -- The general idea is that an Ampersand declaration such as:
   14 --     PURPOSE RELATION r[A*B] IN ENGLISH
   15 --     {+This text explains why r[A*B] exists-}
   16 -- produces the exact right text in the functional specification

   18 -- The class Motivated exists so that we can write the Haskell expression 'purposeOf fSpec l x'
   19 -- anywhere we like for every type of x that could possibly be motivated in an Purpose.
   20 -- 'purpose fSpec l x' produces all explanations related to x from the context (fSpec)
   21 --  that are available in the language specified in 'l'.
   22 -- The other functions in this class are solely meant to be used in the definition of purpose.
   23 -- They are defined once for each instance of Explainable, not be used in other code.
   24 -- TODO: Han, kan dat worden afgeschermd, zodat de programmeur alleen 'purpose' ziet en de andere functies
   25 --       dus niet kan gebruiken?
   26 --     @Stef: Ja, het is al zoveel mogelijk afgeschermd (zie definities die deze module exporteert, hierboven)
   27 --     maar er wordt nog gebruik van gemaakt voor oa foutmeldingen in de atlas, en het prototype.
   28 --     Zodra iemand iets anders verzint voor het gebruik van "ExplainOutputFormat(..),format",
   29 --     kunnen deze uit de export-list van deze module worden verwijderd.
   30 class  Named a => Motivated a where
   31   purposeOf :: FSpec -> Lang -> a -> Maybe [Purpose] -- ^ explains the purpose of a, i.e. the reason why a exists. The purpose could be either given by the user, or generated by Ampersand.
   32                                                     --   Multiple purposes are allowed for the following reasons:
   33                                                     --  * Different purposes from different sources make me want to document them all.
   34                                                     --  * Combining two overlapping scripts from (i.e. from different authors) may cause multiple purposes.
   35   purposeOf fSpec l x = case expls of
   36                         []  -> Nothing -- fatal 40 "No purpose is generated! (should be automatically generated and available in FSpec.)"
   37                         ps  -> Just ps
   38 
   39    where expls = [e | e<-explanations fSpec
   40                     , explForObj x (explObj e)                  -- informally: "if x and e are the same"
   41                     , markupMatchesLang (explMarkup e)
   42                  ]
   43          markupMatchesLang m = amLang m == l
   44   explForObj :: a -> ExplObj -> Bool    -- ^ Given an Explainable object and an ExplObj, return TRUE if they concern the identical object.
   45   explanations :: a -> [Purpose]  -- ^ The explanations that are defined inside a (including that of a itself)
   46   purposesDefinedIn :: FSpec -> Lang -> a -> [Purpose]  -- ^ The explanations that are defined inside a (including that of a itself)
   47   purposesDefinedIn fSpec l x
   48    = [e | e<-explanations fSpec
   49         , amLang (explMarkup e) == l
   50         , explForObj x (explObj e)                  -- informally: "if x and e are the same"
   51      ]

  334 class Meaning a where
  335   meaning :: Lang -> a -> Maybe A_Markup
  336   meaning2Blocks :: Lang -> a -> [Block]
  337   meaning2Blocks l a = case meaning l a of
  338                          Nothing -> []
  339                          Just m  -> amPandoc m

  341 instance Meaning Rule where
  342   meaning l r = case filter isLang (ameaMrk (rrmean r)) of
  343                   []   -> Nothing
  344                   [m]  -> Just m
  345                   _    -> fatal 381 ("In the "++show l++" language, too many meanings given for rule "++name r ++".")
  346                   where isLang m = l == amLang m
  347 
  348 instance Meaning Declaration where
  349   meaning l d =
  350     case d of
  351       Sgn{} -> let isLang m = l == amLang m
  352                in case filter isLang (ameaMrk (decMean d)) of
  353                     []   -> Nothing
  354                     [m]  -> Just m
  355                     _    -> fatal 388 ("In the "++show l++" language, too many meanings given for declaration "++name d ++".")
  356       Isn{} -> fatal 370 "meaning is undefined for Isn"
  357       Vs{}  -> fatal 371 "meaning is undefined for Vs"
  358
