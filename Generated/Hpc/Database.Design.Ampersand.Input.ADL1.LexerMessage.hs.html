   47 data LexerError = LexerError FilePos LexerErrorInfo
   48     deriving(Show)
   49 
   50 data LexerErrorInfo
   51     = UnterminatedComment
   52         | UnterminatedPurpose
   53     | UnterminatedAtom
   54     | UnexpectedChar Char
   55     | NonTerminatedString String
   56     | TooManyClose Char
   57         -- In UnexpectedClose, first char is the closing bracket we see, 
   58         -- second char is the closing bracket we would like to see first
   59         -- e.g. [(1,3]  =>  UnexpectedClose ']' ... ')'
   60     | UnexpectedClose Char FilePos Char
   61     | StillOpenAtEOF [(FilePos, Char)]
   62     deriving(Show)
   63 
   64 showLexerErrorInfo :: LexerErrorInfo -> [String]
   65 showLexerErrorInfo info =
   66     case info of
   67         UnterminatedComment          -> [ Texts.lexerUnterminatedComment               ]
   68         UnterminatedPurpose          -> [ Texts.lexerUnterminatedPurpose               ]
   69         UnterminatedAtom             -> [ Texts.lexerUnterminatedAtom                  ]
   70         UnexpectedChar c             -> [ Texts.lexerUnexpectedChar c                  ]
   71         NonTerminatedString _        -> [ Texts.lexerNonTerminatedString, correctStrings ]
   72         TooManyClose c                 -> [ Texts.lexerTooManyClose c ]
   73         UnexpectedClose c1 _ c2        ->   Texts.lexerUnexpectedClose c1 c2
   74         StillOpenAtEOF [b]             -> [ Texts.lexerStillOpenAtEOF [ show (snd b) ] ]
   75         StillOpenAtEOF bs              -> [ Texts.lexerStillOpenAtEOF (reverse (map (show.snd) bs)) ]
   76             -- 'reverse' because positions will be sorted and brackets are
   77             -- reported in reversed order
   78 
   79 correctStrings :: String
   80 correctStrings = Texts.lexerCorrectStrings

   94 data LexerWarning =
   95     LexerWarning FilePos LexerWarningInfo
   96 
   97 data LexerWarningInfo 
   98     = TabCharacter
   99     | LooksLikeFloatNoFraction String
  100     | LooksLikeFloatNoDigits String
  101     | NestedComment FilePos
  102     | UtfChar
  103     | CommentOperator
  104 
  105 showLexerWarnings :: [LexerWarning] -> String
  106 showLexerWarnings ws = intercalate "\n-----------\n" $ map showWarning ws
  107             where showWarning (LexerWarning pos info) = "Warning: " ++
  108                     intercalate "\n" (showLexerWarningInfo info) ++ " " ++ show pos
  109 
  110 showLexerWarningInfo :: LexerWarningInfo -> [String]
  111 showLexerWarningInfo info = 
  112     case info of
  113         TabCharacter                    -> Texts.lexerTabCharacter
  114         LooksLikeFloatNoFraction digits -> Texts.lexerLooksLikeFloatNoFraction digits
  115         LooksLikeFloatNoDigits fraction -> Texts.lexerLooksLikeFloatNoDigits fraction
  116         NestedComment _                 -> Texts.lexerNestedComment
  117         UtfChar                         -> Texts.lexerUtfChar
  118         CommentOperator                 -> Texts.lexerCommentOperator
  119 
  120 -- TODO: This is only valid for haskell.. Probably more of the warnings too!
  121 keepOneTabWarning :: [LexerWarning] -> [LexerWarning]
  122 keepOneTabWarning = keepOneTab True
  123   where
  124     keepOneTab isFirst (warning@(LexerWarning _ TabCharacter):rest)
  125         | isFirst   = warning : keepOneTab False rest
  126         | otherwise = keepOneTab isFirst rest
  127     keepOneTab isFirst (warning:rest) = 
  128         warning : keepOneTab isFirst rest
  129     keepOneTab _ [] = []
  130 
  131 isLooksLikeFloatWarningInfo :: LexerWarningInfo -> Bool
  132 isLooksLikeFloatWarningInfo warningInfo =
  133    case warningInfo of
  134       LooksLikeFloatNoFraction _ -> True
  135       LooksLikeFloatNoDigits _   -> True
  136       _                          -> False
