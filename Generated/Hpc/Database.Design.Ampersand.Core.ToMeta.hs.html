    8 -- | When dealing with meta-stuff for Ampersand, (Like makeGenerics, makeRAP), 
    9 --   the names of Concepts should be different than 'normal', user-defined Concepts. 
   10 --   This function modifies everything in the context to reflect that.  
   11 toMeta :: MakeMeta a => Options -> (a -> a)
   12 toMeta opts = 
   13   if metaTablesHaveUnderscore opts then makeMeta (string2Meta opts) else id
   14 
   15 string2Meta :: Options -> String -> String
   16 string2Meta opts str 
   17               = if metaTablesHaveUnderscore opts 
   18                 then show ("__"++unquoted++"__")
   19                 else str
   20   where
   21     unquoted
   22      | length str < 2 = str
   23      | head str == '"' && last str == '"' = reverse . tail . reverse . tail $ str
   24      | otherwise = str 
   25 
   26 class MakeMeta a where
   27   makeMeta :: (String -> String) -> a -> a
   28 
   29 instance MakeMeta a => MakeMeta (Maybe a) where
   30   makeMeta _ Nothing = Nothing
   31   makeMeta f (Just a) = Just (makeMeta f a) 
   32 instance MakeMeta a => MakeMeta [a] where
   33   makeMeta _ [] = []
   34   makeMeta f xs = map (makeMeta f) xs

  274 instance MakeMeta a => MakeMeta (Term a) where
  275   makeMeta f t
  276    = case t of
  277        Prim a     -> Prim (makeMeta f a)
  278        PEqu o a b -> PEqu o (makeMeta f a) (makeMeta f b)
  279        PImp o a b -> PImp o (makeMeta f a) (makeMeta f b)
  280        PIsc o a b -> PIsc o (makeMeta f a) (makeMeta f b)
  281        PUni o a b -> PUni o (makeMeta f a) (makeMeta f b)
  282        PDif o a b -> PDif o (makeMeta f a) (makeMeta f b)
  283        PLrs o a b -> PLrs o (makeMeta f a) (makeMeta f b)
  284        PRrs o a b -> PRrs o (makeMeta f a) (makeMeta f b)
  285        PDia o a b -> PDia o (makeMeta f a) (makeMeta f b)
  286        PCps o a b -> PCps o (makeMeta f a) (makeMeta f b)
  287        PRad o a b -> PRad o (makeMeta f a) (makeMeta f b)
  288        PPrd o a b -> PPrd o (makeMeta f a) (makeMeta f b)
  289        PKl0 o a   -> PKl0 o (makeMeta f a)
  290        PKl1 o a   -> PKl1 o (makeMeta f a)
  291        PFlp o a   -> PFlp o (makeMeta f a)
  292        PCpl o a   -> PCpl o (makeMeta f a)
  293        PBrk o a   -> PBrk o (makeMeta f a)
