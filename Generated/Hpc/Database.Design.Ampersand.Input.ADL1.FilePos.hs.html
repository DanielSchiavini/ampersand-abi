   13 --fatal :: Int -> String -> a
   14 --fatal = fatalMsg "Input.ADL1.FilePos"
   15 
   16 type Line = Int
   17 type Column = Int
   18 type SymbolName = String
   19 
   20 initPos :: FilePath -> FilePos
   21 initPos fn = FilePos fn 1 1
   22 
   23 updatePos :: FilePos -> Char -> FilePos
   24 updatePos pos '\n' = addLn 1 pos
   25 updatePos pos '\t' = addTab pos
   26 updatePos pos   _  = addPos 1 pos
   27 
   28 addTab :: FilePos -> FilePos
   29 addTab pos@(FilePos _ _ col) = addPos tabWidth pos
   30     where tabWidth = 8 - ((col-1) `mod` 8)
   31 
   32 addPos :: Int -> FilePos -> FilePos 
   33 addPos n (FilePos fn line col) = FilePos fn line (col + n)
   34 
   35 addLn :: Int -> FilePos -> FilePos 
   36 addLn n (FilePos fn line _) = FilePos fn (line+n) 1
   37 
   38 --Traced a have an origin, which may be unknown.
   39 data FilePos = FilePos FilePath Line Column deriving (Eq, Ord, Generic)
   40 
   41 instance Hashable FilePos where
   42   hashWithSalt s (FilePos fn l c) = s `hashWithSalt` fn `hashWithSalt` l `hashWithSalt` c
   43 
   44 data Origin = OriginUnknown | Origin String | FileLoc FilePos SymbolName | DBLoc String
   45     deriving (Eq, Ord, Typeable, Generic)
   46 
   47 instance Unique Origin where
   48   showUnique = show
   49 instance Hashable Origin
   50 
   51 instance Show FilePos where
   52   show (FilePos fn l c) = "line " ++ show l ++ ":" ++ show c ++ ", file " ++ fn
   53 
   54 instance Show Origin where
   55   show (FileLoc pos _) = show pos
   56   show (DBLoc str)     = "Database location: "++str
   57   show (Origin str)    = str
   58   show OriginUnknown   = "Unknown origin"
   59 
   60 class Traced a where
   61   origin :: a -> Origin
   62   filenm :: a -> String
   63   linenr :: a -> Int
   64   colnr :: a -> Int
   65   filenm x = case origin x of
   66                FileLoc (FilePos nm _ _) _ -> nm
   67                _ -> ""
   68   linenr x = case origin x of
   69                FileLoc (FilePos _ l _) _ -> l
   70                _ -> 0
   71   colnr x  = case origin x of
   72                FileLoc (FilePos _ _ c) _ -> c
   73                _ -> 0
   74 
   75 instance Traced Origin where
   76   origin = id
