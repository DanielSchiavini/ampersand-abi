    6 -- | Converts the first character of a string to lowercase, with the exception that there is a second character, which is uppercase.
    7 -- uncap "AbcDe" == "abcDe"
    8 -- uncap "ABcDE" == "ABcDE"
    9 unCap :: String -> String
   10 unCap [] = []
   11 unCap [h] = [toLower h]
   12 unCap (h:h':t) | isUpper h' = h:h':t
   13                | otherwise  = toLower h:h':t
   14 -- | Converts the first character of a string to uppercase
   15 upCap :: String -> String
   16 upCap [] = []
   17 upCap (h:t) = toUpper h:t
   18 
   19 -- | escape anything except regular characters and digits to _<character code>
   20 -- e.g. escapeNonAlphaNum "a_Ã©" = "a_95_233"
   21 escapeNonAlphaNum :: String -> String
   22 escapeNonAlphaNum = concatMap escapeNonAlphaNumChar
   23  where escapeNonAlphaNumChar c
   24          | isAlphaNum c && isAscii c = [c]
   25          | otherwise                 = '_' : show (ord c)
