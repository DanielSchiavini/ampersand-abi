   40 getDataFileName :: FilePath -> IO FilePath
   41 getDataFileName name = do
   42   dir <- getDataDir
   43   return (dir `joinFileName` name)
   44 
   45 getPrefixDirRel :: FilePath -> IO FilePath
   46 getPrefixDirRel dirRel = try_size 2048 -- plenty, PATH_MAX is 512 under Win32.
   47   where
   48     try_size size = allocaArray (fromIntegral size) $ \buf -> do
   49         ret <- c_GetModuleFileName nullPtr buf size
   50         case ret of
   51           0 -> return (prefix `joinFileName` dirRel)
   52           _ | ret < size -> do
   53               exePath <- peekCWString buf
   54               let (bindir,_) = splitFileName exePath
   55               return ((bindir `minusFileName` bindirrel) `joinFileName` dirRel)
   56             | otherwise  -> try_size (size * 2)
   57

   61 minusFileName :: FilePath -> String -> FilePath
   62 minusFileName dir ""     = dir
   63 minusFileName dir "."    = dir
   64 minusFileName dir suffix =
   65   minusFileName (fst (splitFileName dir)) (fst (splitFileName suffix))
   66 
   67 joinFileName :: String -> String -> FilePath
   68 joinFileName ""  fname = fname
   69 joinFileName "." fname = fname
   70 joinFileName dir ""    = dir
   71 joinFileName dir fname
   72   | isPathSeparator (last dir) = dir++fname
   73   | otherwise                  = dir++pathSeparator:fname
   74 
   75 splitFileName :: FilePath -> (String, String)
   76 splitFileName p = (reverse (path2++drive), reverse fname)
   77   where
   78     (path,drive) = case p of
   79        (c:':':p') -> (reverse p',[':',c])
   80        _          -> (reverse p ,"")
   81     (fname,path1) = break isPathSeparator path
   82     path2 = case path1 of
   83       []                           -> "."
   84       [_]                          -> path1   -- don't remove the trailing slash if 
   85                                               -- there is only one character
   86       (c:path') | isPathSeparator c -> path'
   87       _                             -> path1
   88 
   89 pathSeparator :: Char
   90 pathSeparator = '\\'
   91 
   92 isPathSeparator :: Char -> Bool
   93 isPathSeparator c = c == '/' || c == '\\'
