   22 import Control.Applicative
   23 import Control.Monad
   24 
   25 type Bracket = (FilePos, Char)
   26 
   27 -- Output monad: [LexerWarning]
   28 -- State monad: FilePos and [Bracket]
   29 newtype LexerMonad a =
   30     LM ([Options] -> FilePos -> [Bracket] ->
   31         Either LexerError (a, [LexerWarning], FilePos, [Bracket]))
   32 
   33 unLM :: LexerMonad t -> [Options] -> FilePos -> [Bracket]
   34           -> Either LexerError (t, [LexerWarning], FilePos, [Bracket])
   35 unLM (LM x) = x
   36 
   37 bindLM :: LexerMonad a -> (a -> LexerMonad b) -> LexerMonad b
   38 bindLM (LM f) g =
   39     LM (\opts pos brackets ->
   40         case f opts pos brackets of
   41             Left err -> Left err
   42             Right (a, warnings, pos2, brackets2) ->
   43                 case unLM (g a) opts pos2 brackets2 of
   44                     Left err -> Left err
   45                     Right (b, moreWarnings, pos3, brackets3) ->
   46                         Right (b, warnings ++ moreWarnings, pos3, brackets3))
   47 
   48 returnLM :: a -> LexerMonad a
   49 returnLM x = LM (\_ pos brackets -> Right (x, [], pos, brackets))
   50 
   51 instance Monad LexerMonad where
   52     (>>=) = bindLM
   53     return = returnLM
   54 
   55 instance Functor LexerMonad where
   56     -- fmap :: (a -> b) -> LexerMonad a -> LexerMonad b
   57     fmap ab la = do { a <- la; return (ab a) }
   58 
   59 instance Applicative LexerMonad where
   60     pure = returnLM
   61     (<*>) = ap
   62 
   63 runLexerMonad :: [Options] -> FilePath -> LexerMonad a -> Either LexerError (a, [LexerWarning])
   64 runLexerMonad opts file (LM f) =
   65     case f opts (initPos file) [] of
   66         Left err -> Left err
   67         Right (a, warnings, _, _) -> Right (a, keepOneTabWarning warnings)
   68 
   69 -- TODO: These methods are not being used anywhere
   70 getOpts :: LexerMonad [Options]
   71 getOpts = LM (\opts pos brackets -> Right (opts, [], pos, brackets))
   72 
   73 getPos :: LexerMonad FilePos
   74 getPos = LM (\_ pos brackets -> Right (pos, [], pos, brackets))
   75 
   76 incPos :: Int -> LexerMonad ()
   77 incPos i = LM (\_ pos brackets -> Right ((), [], addPos i pos, brackets))
   78 
   79 nextPos :: Char -> LexerMonad ()
   80 nextPos c = LM (\_ pos brackets -> Right ( (), [], updatePos pos c, brackets ))
   81 
   82 lexerError :: LexerErrorInfo -> FilePos -> LexerMonad a
   83 lexerError err pos =
   84     LM (\_ _ _ -> Left (LexerError pos err))
   85 
   86 lexerWarning :: LexerWarningInfo -> FilePos -> LexerMonad ()
   87 lexerWarning warning warningPos =
   88     LM (\_ pos brackets ->
   89         Right ((), [LexerWarning warningPos warning], pos, brackets))
   90 
   91 openBracket :: Char -> LexerMonad ()
   92 openBracket c = LM (\_ pos brackets ->
   93     Right ( (), [], pos, (pos, c) : brackets ))
   94 
   95 closeBracket :: Char -> LexerMonad ()
   96 closeBracket c = LM (\_ pos brackets ->
   97     case brackets of
   98         [] -> Left (LexerError pos (TooManyClose c))
   99         (pos2, c2):rest
  100             | matchBracket c2 c ->
  101                 Right ((), [], pos, rest)
  102             | otherwise ->
  103                 Left (LexerError pos (UnexpectedClose c pos2 c2))
  104     )
  105 
  106 checkBracketsAtEOF :: LexerMonad ()
  107 checkBracketsAtEOF = LM (\_ pos brackets ->
  108     case brackets of
  109         [] -> Right ((), [], pos, [])
  110         _  -> Left (LexerError pos (StillOpenAtEOF brackets))
  111     )
  112 
  113 matchBracket :: Char -> Char -> Bool
  114 matchBracket '(' ')' = True
  115 matchBracket '[' ']' = True
  116 matchBracket '{' '}' = True
  117 matchBracket _ _ = False
