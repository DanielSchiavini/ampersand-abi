   35 select :: IORef Language -> [Arrow Language msg] -> msg
   36 select languageRef table =
   37     let lang = unsafePerformIO (readIORef languageRef)
   38         convert (a :-> b) = (a, b)
   39     in fromMaybe (error "Texts.select: unknown language")
   40         (lookup lang (map convert table))

  445 commasAnd :: [String] -> String
  446 commasAnd [] = []
  447 commasAnd [x] = x
  448 commasAnd (x:xs) = x ++ concatMap (", " ++) (init xs) ++ " and " ++ last xs
  449 
  450 kommasEn :: [String] -> String
  451 kommasEn [] = []
  452 kommasEn [x] = x
  453 kommasEn (x:xs) = x ++ concatMap (", " ++) (init xs) ++ " en " ++ last xs
