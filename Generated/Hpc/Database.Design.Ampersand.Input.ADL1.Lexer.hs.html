   94 mainLexer :: Lexer
   95 
   96 -----------------------------------------------------------
   97 -- Removing unnecessary text artifacts (comment, spaces,...)
   98 -----------------------------------------------------------
   99 
  100 mainLexer _ [] =  return []
  101 
  102 mainLexer p ('-':'-':s) = mainLexer p (skipLine s) --TODO: Test if we should increase line number and reset the column number
  103 
  104 mainLexer p (c:s) | isSpace c = let (spc,next) = span isSpace s
  105                                 in  mainLexer (foldl updatePos p (c:spc)) next
  106 
  107 mainLexer p ('-':'+':s)  = returnToken lx p mainLexer p rest
  108                 where lx   = LexExpl $ dropWhile isSpace (takeLine s)
  109                       rest = skipLine s
  110 
  111 mainLexer p ('{':'-':s) = lexNest mainLexer (addPos 2 p) s
  112 mainLexer p ('{':'+':s) = lexExpl mainLexer (addPos 2 p) s
  113 mainLexer p ('"':ss) =
  114     let (s,swidth,rest) = scanString ss
  115     in if null rest || head rest /= '"'
  116                               then lexerError (NonTerminatedString s) p
  117                               else returnToken (LexString s) p mainLexer (addPos (swidth+2) p) (tail rest)
  118 
  119 {- In Ampersand, atoms may be promoted to singleton relations by single-quoting them. For this purpose, we treat
  120    single quotes exactly as the double quote for strings. That substitutes the scanner code for character literals. -}
  121 mainLexer p ('\'':ss)
  122      = let (s,swidth,rest) = scanAtom ss
  123        in if null rest || head rest /= '\''
  124              then lexerError UnterminatedAtom p
  125              else returnToken (LexAtom s) p mainLexer (addPos (swidth+2) p) (tail rest)
  126 
  127 -----------------------------------------------------------
  128 -- looking for keywords - operators - special chars
  129 -----------------------------------------------------------
  130 
  131 -- Special case for < since it's the beginning of operators but also a symbol when alone
  132 mainLexer p ('<':d:s) = if isOperator ['<',d]
  133                         then returnToken (LexOperator ['<',d]) p mainLexer (addPos 2 p) s
  134                         else returnToken (LexSymbol    '<')    p mainLexer (addPos 1 p) (d:s)
  135 
  136 mainLexer p cs@(c:s)
  137      | isIdStart c || isUpper c
  138          = let (name', p', s')    = scanIdent (addPos 1 p) s
  139                name               = c:name'
  140                tokt   | iskw name = LexKeyword name
  141                       | otherwise = if isIdStart c
  142                                     then LexVarId name
  143                                     else LexConId name
  144            in returnToken tokt p mainLexer p' s'
  145      | isOperatorBegin c
  146          = let (name, s') = getOp cs
  147            in returnToken (LexOperator name) p mainLexer (foldl updatePos p name) s'
  148      | isSymbol c = returnToken (LexSymbol c) p mainLexer (addPos 1 p) s
  149      | isDigit c
  150          = let (tk,_,width,s') = getNumber cs
  151            in  returnToken tk p mainLexer (addPos width p) s'
  152        -- Ignore unexpected characters in the beginning of the file because of the UTF-8 BOM marker.
  153        -- TODO: Find out the right way of handling the BOM marker.
  154      | beginFile p = do { lexerWarning UtfChar p; mainLexer p s }
  155      | otherwise  = lexerError (UnexpectedChar c) p
  156     where beginFile (FilePos _ 1 1) = True
  157           beginFile _ = False

  169 locatein :: Ord a => [a] -> a -> Bool
  170 locatein es = isJust . btLocateIn compare (tab2tree (sort es))
  171 
  172 iskw :: String -> Bool
  173 iskw = locatein keywords
  174 
  175 isSymbol :: Char -> Bool
  176 isSymbol = locatein symbols
  177 
  178 isOperator :: String -> Bool
  179 isOperator  = locatein operators
  180 
  181 isOperatorBegin :: Char -> Bool
  182 isOperatorBegin  = locatein (map head operators)
  183 
  184 isIdStart :: Char -> Bool
  185 isIdStart c = isLower c || c == '_'
  186 
  187 isIdChar :: Char -> Bool
  188 isIdChar c =  isAlphaNum c || c == '_'
  189 
  190 -- Finds the longest prefix of cs occurring in keywordsops
  191 getOp :: String -> (String, String)
  192 getOp cs = findOper operators cs ""
  193     where findOper :: [String] -> String -> String -> (String,String)
  194           findOper []  _     _  = ("", cs)
  195           findOper _   []    op = (op,[])
  196           findOper ops (c:rest) op =
  197               if null found then (op, c:rest)
  198               else findOper found rest (op ++ [c])
  199               where found = [s' | o:s'<-ops, c==o]
  200 
  201 -- scan ident receives a file position and the resting contents, returning the scanned identifier, the file location and the resting contents.
  202 scanIdent :: FilePos -> String -> (String, FilePos, String)
  203 scanIdent p s = let (name,rest) = span isIdChar s
  204                 in (name,addPos (length name) p,rest)

  207 scanAtom :: String -> (String,Int,String)
  208 scanAtom []              = ("",0,[])
  209 scanAtom ('\\':'&':xs)   = let (str,w,r) = scanAtom xs
  210                            in (str,w+2,r)
  211 scanAtom ('"':xs)        = let (str,w,r) = scanAtom xs
  212                            in ('"': str,w+1,r)
  213 scanAtom xs   = let (ch,cw,cr) = getchar xs
  214                     (str,w,r)  = scanAtom cr
  215                 in maybe ("",0,xs) (\c -> (c:str,cw+w,r)) ch

  218 -- String clean-up functions / comments

  221 lexNest :: Lexer -> Lexer
  222 lexNest c p ('-':'}':s) = c (addPos 2 p) s
  223 lexNest c p ('{':'-':s) = lexNest (lexNest c) (addPos 2 p) s
  224 lexNest c p (x:s)       = lexNest c (updatePos p x) s
  225 lexNest _ p []          = lexerError UnterminatedComment p
  226 
  227 --TODO: Also accept {+ ... +} as delimiters
  228 lexExpl :: Lexer -> Lexer
  229 lexExpl = lexExpl' ""
  230  where lexExpl' str _ p ('-':'}':s) = returnToken (LexExpl str) p mainLexer (addPos 2 p)  s
  231        lexExpl' str c p ('{':'-':s) = lexNest (lexExpl' str c) (addPos 2 p) s
  232        lexExpl' str c p ('-':'-':s) = lexExpl' str c p (dropWhile (/= '\n') s)
  233        lexExpl' str c p (x:s)       = lexExpl' (str++[x]) c (updatePos p x) s
  234        lexExpl' _   _ p []          = lexerError UnterminatedPurpose p

  240 -- Returns tuple with the parsed lexeme, the integer, the amount of read characters and the rest of the text
  241 getNumber :: String -> (Lexeme, Int, Int, String)
  242 getNumber [] = fatal 294 "getNumber"
  243 getNumber cs@(c:s)
  244   | c /= '0'         = num10
  245   | null s           = const0
  246   | hs `elem` "xX"   = num16
  247   | hs `elem` "oO"   = num8
  248   | otherwise        = num10
  249   where (hs:ts) = s
  250         const0 = (LexDecimal 0, 0, 1, s)
  251         num10  = let (n, rs) = span isDigit cs
  252                      nr = read n
  253                  in (LexDecimal nr, nr, length n, rs)
  254         num16   = readNum isHexaDigit  16 LexHex
  255         num8    = readNum isOctDigit 8  LexOctal
  256         readNum :: (Char -> Bool) -> Int -> (Int -> Lexeme) -> (Lexeme, Int, Int, String)
  257         readNum p base lx
  258           = let (n, rs) = span p ts
  259             in  if null n
  260                 then const0
  261                 else let nr = readn base n
  262                      in (lx nr, nr, 2 + length n, rs)
  263 
  264 isHexaDigit :: Char -> Bool
  265 isHexaDigit  d = isDigit d || (d >= 'A' && d <= 'F') || (d >= 'a' && d <= 'f')
  266 
  267 value :: Char -> Int
  268 value c | isDigit c = ord c - ord '0'
  269         | isUpper c = ord c - ord 'A' + 10
  270         | isLower c = ord c - ord 'a' + 10
  271         | otherwise = fatal 321 ("value undefined for '"++ show c++"'")

  277 scanString :: String -> (String, Int, String)
  278 scanString []            = ("",0,[])
  279 scanString ('\\':'&':xs) = let (str,w,r) = scanString xs  -- TODO: why do we ignore \& ?
  280                            in (str,w+2,r)
  281 scanString ('\\':'\'':xs) = let (str,w,r) = scanString xs -- escaped single quote: \'  (redundant, but allowed in most languages, and it makes escaping generated code a lot easier.)
  282                            in ('\'': str,w+2,r)
  283 scanString ('\'':xs)     = let (str,w,r) = scanString xs  -- single quote: '
  284                            in ('\'': str,w+1,r)
  285 scanString xs = let (ch,cw,cr) = getchar xs
  286                     (str,w,r)  = scanString cr
  287                 in maybe ("",0,xs) (\c -> (c:str,cw+w,r)) ch
  288 
  289 getchar :: String -> (Maybe Char, Int, String)
  290 getchar []          = (Nothing,0,[])
  291 getchar s@('\n':_ ) = (Nothing,0,s)
  292 getchar s@('\t':_ ) = (Nothing,0,s)
  293 getchar s@('\'':_ ) = (Nothing,0,s)
  294 getchar s@('"' :_ ) = (Nothing,0,s)
  295 getchar   ('\\':xs) = let (c,l,r) = getEscChar xs
  296                       in (c,l+1,r)
  297 getchar (x:xs)      = (Just x,1,xs)
  298 
  299 getEscChar :: String -> (Maybe Char, Int, String)
  300 getEscChar [] = (Nothing,0,[])
  301 getEscChar s@(x:xs) | isDigit x = let (_, val, len, rest) = getNumber s
  302                                   in  if val >= 0 && val <= 255
  303                                          then (Just (chr val),len, rest)
  304                                          else (Nothing, 1, rest)
  305                     | otherwise = case x `lookup` cntrChars of
  306                                  Nothing -> (Nothing,0,s)
  307                                  Just c  -> (Just c,1,xs)
  308   where cntrChars = [('a','\a'),('b','\b'),('f','\f'),('n','\n'),('r','\r'),('t','\t')
  309                     ,('v','\v'),('\\','\\'),('"','\"')]
  310 
  311 readn :: Int -> String -> Int
  312 readn base = foldl (\r x  -> value x + base * r) 0

  318 returnToken :: Lexeme -> FilePos -> Lexer -> Lexer
  319 returnToken lx pos continue posi input = do
  320     let token = Tok lx pos
  321     tokens <- continue posi input
  322     return (token:tokens)
